LeetCode第59题螺旋矩阵2-C++

Created at 2019-03-12 Updated at 2019-03-12 Category C++ / Leetcode Tag C++ / LeetCode

题目

思路

class Solution {
public:
    vector<vector<int>> generateMatrix(int n) {
        vector<int> row(n, 0);
        vector<vector<int>> res(n, row);

        int up = 0, down = n - 1, left = 0, right = n - 1;
        int count = 1;
        while(count <= n*n){
            for(int j = left; j <= right; j++){
                res[up][j] = count;
                count++;
            }
            for(int i = ++up; i <= down; i++){
                res[i][right] = count;
                count++;
            }
            for(int j = --right; j >= left; j--){
                res[down][j] = count;
                count++;
            }
            for(int i = --down; i >= up; i--){
                res[i][left] = count;
                count++;
            }
            left++;
        }
        return res;
    }
};
