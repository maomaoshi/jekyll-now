LeetCode第35题搜索插入位置-C++

撰写于 2019-02-08 修改于 2019-02-08 分类 C++ / Leetcode 标签 C++ / LeetCode

题目

使用二分搜索法，若找到就返回索引值。若没有找到，最终left=right,且left就是要插入的位置。

class Solution {
public:
    int searchInsert(vector<int>& nums, int target) {
        if(nums. size() == 0){
            nums.push_back(target);
            return 0;
        }
        int left = 0, right = nums.size();
        int mid;
        while(left < right){
            mid = (left + right) / 2;
            if(nums[mid] == target){
                return mid;
            }
            else if(nums[mid] > target){
                right = mid;
            }
            else{
                left = mid + 1;
            }
        }
        nums.insert(nums.begin()+left, target);
        return left;
    }
};
